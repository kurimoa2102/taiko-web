<body>

    <div class="container">
        <h1>YouTube Privacy Player</h1>
        <p>YouTubeの動画URLを貼り付けて「表示」ボタンを押してください。<br>かしわ制作のプライバシー強化モード動画を再生します。</p>

        <div class="input-group">
            <input type="text" id="youtubeUrl" placeholder="ここにYouTubeのURLを貼り付け">
            <button id="playButton">表示</button>
        </div>

        <div class="video-wrapper" id="videoContainer">
            <!-- 動画はここに埋め込まれます -->
        </div>
        <p id="errorMessage"></p>
    </div>

    <script>
        // HTMLの要素を取得
        const urlInput = document.getElementById('youtubeUrl');
        const playButton = document.getElementById('playButton');
        const videoContainer = document.getElementById('videoContainer');
        const errorMessage = document.getElementById('errorMessage');

        // ボタンがクリックされたときの処理
        playButton.addEventListener('click', function() {
            const originalUrl = urlInput.value;
            errorMessage.textContent = ''; // エラーメッセージをリセット
            
            // 入力されたURLから動画IDを抽出
            const videoId = extractVideoId(originalUrl);

            if (videoId) {
                // no-cookie版の埋め込みURLを生成
                const embedUrl = `https://www.youtube-nocookie.com/embed/${videoId}`;
                
                // iframeを作成して動画を表示
                videoContainer.innerHTML = `<iframe src="${embedUrl}" title="YouTube video player" frameborder="0" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>`;
            } else {
                // 有効なURLでない場合はエラーメッセージを表示
                errorMessage.textContent = '有効なYouTubeのURLを入力してください。';
                videoContainer.innerHTML = ''; // 動画表示エリアを空にする
            }
        });

        // URLからYouTubeの動画IDを抽出する関数
        function extractVideoId(url) {
            if (!url) {
                return null;
            }
            // 様々な形式のYouTube URLに対応するための正規表現
            const regex = /(?:https?:\/\/)?(?:www\.)?(?:youtube\.com\/(?:watch\?v=|embed\/|v\/)|youtu\.be\/)([\w-]{11})/;
            const match = url.match(regex);
            
            return match ? match[1] : null;
        }
